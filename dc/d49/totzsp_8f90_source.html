34 USE vmec_main

56 ! WORK1 Array of inverse transforms in toroidal angle (zeta), for all radial positions
57 ! NOTE: ORDERING OF LAST INDEX IS DIFFERENT HERE THAN IN PREVIOUS VMEC2000 VERSIONS
58 REAL(rprec), DIMENSION(:,:), ALLOCATABLE :: work1
59 REAL(rprec) :: cosmux, sinmux

61 ! This is the inverse Fourier transform from the (odd-m-scaled-by-1/sqrt(s))-xc
62 ! given in rzl_array to real space. Tangential derivatives are also computed.
63
64 ! CONVERT FROM INTERNAL XC REPRESENTATION FOR m=1 MODES, R+(stored at rss) = .5(rss + zcs),
65 ! R-(stored at zcs) = .5(rss - zcs), TO EXTERNAL ("PHYSICAL") rss, zcs FORMS. NEED THIS EVEN
66 ! WHEN COMPUTING HESSIAN FOR FREE BOUNDARY (rmnss, zmncs at JS=NS needed in vacuum call)
67 !
68 ! WHEN COMPUTING PRECONDITIONER, USE FASTER HESSIAN VERSION (totzsps_hess) INSTEAD.
69 rmncc => rzl_array(:,:,:,rcc) ! COS(mu) COS(nv)
70 zmnsc => rzl_array(:,:,:,zsc+ntmax) ! SIN(mu) COS(nv)
71 lmnsc => rzl_array(:,:,:,zsc+2*ntmax) ! SIN(mu) COS(nv)
72 IF (lthreed) THEN
73 rmnss => rzl_array(:,:,:,rss) ! SIN(mu) SIN(nv)
74 zmncs => rzl_array(:,:,:,zcs+ntmax) ! COS(mu) SIN(nv)
75 lmncs => rzl_array(:,:,:,zcs+2*ntmax) ! COS(mu) SIN(nv)
76
77! #ifndef _HBANGLE
78 ! CONVERT FROM INTERNAL XC REPRESENTATION FOR m=1 MODES, R+(stored at rss) = .5(rss + zcs),
79 ! R-(stored at zcs) = .5(rss - zcs), TO EXTERNAL ("PHYSICAL") rss, zcs FORMS. NEED THIS EVEN
80 ! WHEN COMPUTING HESSIAN FOR FREE BOUNDARY (rmnss, zmncs at JS=NS needed in vacuum call)
81
82 CALL convert_sym (rmnss, zmncs)
83! #end /* ndef _HBANGLE */
84
85 END IF
86
87 ! v8.50: Norm for preconditioned R,Z forces: scale to boundary value only
88 ! v8.51 Restore hs dependence (1:ns, not just ns)
89 ! fnorm1 = one/SUM(rzl_array(1:ns,:,m1:,1:2*ntmax)**2)
90
91 ! ORIGIN EXTRAPOLATION (JS=1) FOR M=1 MODES
92 ! --> TODO: why? quantities on axis should not have a poloidal dependence ???
93 ! NOTE: PREVIOUS VERSIONS OF VMEC USED TWO-POINT EXTRAPOLATION FOR R,Z.
94 ! HOWEVER,THIS CAN NOT BE USED TO COMPUTE THE TRI-DIAG 2D PRECONDITIONER.
95 rzl_array(1,:,m1,:) = rzl_array(2,:,m1,:) ! now this is a constant extrapolation to the axis (????)
96
97 ioff = lbound(rmncc,2) ! starting index along n
98 joff = lbound(rmncc,3) ! starting index along m
99
100 ! ORIGIN EXTRAPOLATION OF M=0 MODES FOR LAMBDA
101 IF (lthreed .and. jlam(m0).gt.1) then
102 lmncs(1,:,m0+joff) = lmncs(2,:,m0+joff)
103 end if
104
105 nsz = ns*nzeta
106 ALLOCATE (work1(nsz,12), stat=i)
107 IF (i .ne. 0) stop 'Allocation error in VMEC2000 totzsps'
108
109 r11 = 0; ru1 = 0; rv1 = 0; rcn1 = 0
110 z11 = 0; zu1 = 0; zv1 = 0; zcn1 = 0
111 lu1 = 0; lv1 = 0
112
113 ! COMPUTE R, Z, AND LAMBDA IN REAL SPACE
114 ! NOTE: LU = d(Lam)/du, LV = -d(Lam)/dv
115 DO m = 0, mpol1
116 mparity = mod(m,2)
117 mj = m+joff
118 work1 = 0
119
120 ! m>1 contributions start at js=2, not 1 (axis)
121 ! TODO: why has the axis even an m=1 contribution?
122 j1 = jmin1(m)
123
124 ! INVERSE TRANSFORM IN N-ZETA, FOR FIXED M
125 DO n = 0, ntor
126 ni = n+ioff
127 DO k = 1, nzeta
128 l = ns*(k-1)
129
130 j1l = j1+l
131 nsl = ns+l
132
133 work1(j1l:nsl, 1) = work1(j1l:nsl, 1) + rmncc(j1:ns,ni,mj)*cosnv(k,n)
134 work1(j1l:nsl, 6) = work1(j1l:nsl, 6) + zmnsc(j1:ns,ni,mj)*cosnv(k,n)
135 work1(j1l:nsl,10) = work1(j1l:nsl,10) + lmnsc(j1:ns,ni,mj)*cosnv(k,n)
136
137 IF (.not.lthreed) cycle
138
139 work1(j1l:nsl, 4) = work1(j1l:nsl, 4) + rmnss(j1:ns,ni,mj)*cosnvn(k,n)
140 work1(j1l:nsl, 7) = work1(j1l:nsl, 7) + zmncs(j1:ns,ni,mj)*cosnvn(k,n)
141 work1(j1l:nsl,11) = work1(j1l:nsl,11) + lmncs(j1:ns,ni,mj)*cosnvn(k,n)
142
143 work1(j1l:nsl, 2) = work1(j1l:nsl, 2) + rmnss(j1:ns,ni,mj)*sinnv(k,n)
144 work1(j1l:nsl, 5) = work1(j1l:nsl, 5) + zmncs(j1:ns,ni,mj)*sinnv(k,n)
145 work1(j1l:nsl, 9) = work1(j1l:nsl, 9) + lmncs(j1:ns,ni,mj)*sinnv(k,n)
146
147 work1(j1l:nsl, 3) = work1(j1l:nsl, 3) + rmncc(j1:ns,ni,mj)*sinnvn(k,n)
148 work1(j1l:nsl, 8) = work1(j1l:nsl, 8) + zmnsc(j1:ns,ni,mj)*sinnvn(k,n)
149 work1(j1l:nsl,12) = work1(j1l:nsl,12) + lmnsc(j1:ns,ni,mj)*sinnvn(k,n)
150 END DO
151 END DO
152
153 ! INVERSE TRANSFORM IN M-THETA, FOR ALL RADIAL, ZETA VALUES
154 l = 0
155 DO i = 1, ntheta2
156
157 ! start and end indices of poloidal slice of current flux surface
158 j1l = l+1
159 nsl = l+nsz
160
161 l = l + nsz
162
163 ! basis functions for spectral constraint
164 cosmux = xmpq(m,1)*cosmu(i,m)
165 sinmux = xmpq(m,1)*sinmu(i,m)
166
167 r11(j1l:nsl,mparity) = r11(j1l:nsl,mparity) + work1(1:nsz,1)*cosmu(i,m)
168 ru1(j1l:nsl,mparity) = ru1(j1l:nsl,mparity) + work1(1:nsz,1)*sinmum(i,m)
169
170! #ifndef _HBANGLE
171 rcn1(j1l:nsl,mparity) = rcn1(j1l:nsl,mparity) + work1(1:nsz,1)*cosmux ! rmncc*cosnv*cosmux
172! #end /* ndef _HBANGLE */
173
174 z11(j1l:nsl,mparity) = z11(j1l:nsl,mparity) + work1(1:nsz,6)*sinmu(i,m)
175 zu1(j1l:nsl,mparity) = zu1(j1l:nsl,mparity) + work1(1:nsz,6)*cosmum(i,m)
176
177! #ifndef _HBANGLE
178 zcn1(j1l:nsl,mparity) = zcn1(j1l:nsl,mparity) + work1(1:nsz,6)*sinmux ! zmnsc*cosnv*sinmux
179! #end /* ndef _HBANGLE */
180
181 lu1(j1l:nsl,mparity) = lu1(j1l:nsl,mparity) + work1(1:nsz,10)*cosmum(i,m)
182
183 IF (.not.lthreed) cycle
184
185 r11(j1l:nsl,mparity) = r11(j1l:nsl,mparity) + work1(1:nsz,2)*sinmu(i,m)
186 ru1(j1l:nsl,mparity) = ru1(j1l:nsl,mparity) + work1(1:nsz,2)*cosmum(i,m)
187
188! #ifndef _HBANGLE
189 rcn1(j1l:nsl,mparity) = rcn1(j1l:nsl,mparity) + work1(1:nsz,2)*sinmux ! rmnss*sinnv*sinmux
190! #end /* ndef _HBANGLE */
191
192 rv1(j1l:nsl,mparity) = rv1(j1l:nsl,mparity) + work1(1:nsz,3)*cosmu(i,m) &
193 + work1(1:nsz,4)*sinmu(i,m)
194 z11(j1l:nsl,mparity) = z11(j1l:nsl,mparity) + work1(1:nsz,5)*cosmu(i,m)
195
196 zu1(j1l:nsl,mparity) = zu1(j1l:nsl,mparity) + work1(1:nsz,5)*sinmum(i,m)
197
198! #ifndef _HBANGLE
199 zcn1(j1l:nsl,mparity) = zcn1(j1l:nsl,mparity) + work1(1:nsz,5)*cosmux ! zmncs*sinnv*cosmux
200! #end /* ndef _HBANGLE */
201
202 zv1(j1l:nsl,mparity) = zv1(j1l:nsl,mparity) + work1(1:nsz,7)*cosmu(i,m) &
203 + work1(1:nsz,8)*sinmu(i,m)
204
205 lu1(j1l:nsl,mparity) = lu1(j1l:nsl,mparity) + work1(1:nsz,9)*sinmum(i,m)
206 lv1(j1l:nsl,mparity) = lv1(j1l:nsl,mparity) - ( work1(1:nsz,11)*cosmu(i,m) &
207 + work1(1:nsz,12)*sinmu(i,m) )
208 END DO
209
210 END DO
211
212 DEALLOCATE (work1)
